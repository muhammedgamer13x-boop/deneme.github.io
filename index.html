AI Gesture Volume Control (.py)

💡 Bu Program Nasıl Çalıştırılır?

1. Bilgisayarınızda Python yüklü olmalıdır.
2. Terminale şu komutu yazarak gerekli kütüphaneleri yükleyin:
pip install opencv-python mediapipe pycaw numpy
3. Kodları bir dosyaya ses_kontrol.py adıyla kaydedip çalıştırın!

import cv2
import mediapipe as mp
import math
import numpy as np
from ctypes import cast, POINTER
from comtypes import CLSCTX_ALL
from pycaw.pycaw import AudioUtilities, IAudioEndpointVolume

# 1. Kamerayı ve MediaPipe El Algılayıcıyı Başlat
cap = cv2.VideoCapture(0)
mp_hands = mp.solutions.hands
hands = mp_hands.Hands(max_num_hands=1, min_detection_confidence=0.7)
mp_draw = mp.solutions.drawing_utils

# 2. Bilgisayarın Ses Kartı Ayarlarına Bağlan (Windows için)
devices = AudioUtilities.GetSpeakers()
interface = devices.Activate(IAudioEndpointVolume._iid_, CLSCTX_ALL, None)
volume = cast(interface, POINTER(IAudioEndpointVolume))
vol_range = volume.GetVolumeRange() # [Min Ses, Max Ses]
min_vol = vol_range[0]
max_vol = vol_range[1]

print("[SİSTEM] Yapay Zeka Ses Kontrolü Başlatıldı. Çıkış için 'q' basın.")

while True:
    success, img = cap.read()
    if not success:
        break
        
    img = cv2.flip(img, 1) # Aynalama
    h, w, c = img.shape
    img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
    results = hands.process(img_rgb)
    
    vol_bar = 400 # Varsayılan grafik bar yüksekliği
    vol_per = 0   # Varsayılan yüzde seviyesi
    
    if results.multi_hand_landmarks:
        for hand_lms in results.multi_hand_landmarks:
            # El iskeletini çiz
            mp_draw.draw_landmarks(img, hand_lms, mp_hands.HAND_CONNECTIONS)
            
            # Başparmak ucu (Landmark 4) ve İşaret parmağı ucu (Landmark 8) koordinatları
            thumb = hand_lms.landmark[4]
            index = hand_lms.landmark[8]
            
            tx, ty = int(thumb.x * w), int(thumb.y * h)
            ix, iy = int(index.x * w), int(index.y * h)
            
            # Parmak uçlarına daire çiz ve aralarına çizgi çek
            cv2.circle(img, (tx, ty), 10, (0, 240, 255), cv2.FILLED)
            cv2.circle(img, (ix, iy), 10, (0, 240, 255), cv2.FILLED)
            cv2.line(img, (tx, ty), (ix, iy), (0, 240, 255), 3)
            
            # İki parmak arasındaki mesafeyi ölç (Hipotenüs)
            length = math.hypot(ix - tx, iy - ty)
            
            # Algılanan mesafeyi Python Ses Aralığına ve Yüzdeye dönüştür
            # El mesafesi genelde 20px ile 200px arasındadır
            vol = np.interp(length, [20, 200], [min_vol, max_vol])
            vol_bar = np.interp(length, [20, 200], [400, 150])
            vol_per = np.interp(length, [20, 200], [0, 100])
            
            # Bilgisayarın ses seviyesini güncelle
            volume.SetMasterVolumeLevel(vol, None)
            
            # Eğer parmaklar çok yakınsa yeşil sinyal ver
            if length < 25:
                cv2.circle(img, ((tx + ix) // 2, (ty + iy) // 2), 10, (0, 255, 0), cv2.FILLED)

    # Ekrana Görsel Ses Barları ve Grafikler Çiz
    cv2.rectangle(img, (50, 150), (85, 400), (255, 0, 0), 3)
    cv2.rectangle(img, (50, int(vol_bar)), (85, 400), (0, 240, 255), cv2.FILLED)
    cv2.putText(img, f'{int(vol_per)} %', (40, 450), cv2.FONT_HERSHEY_COMPLEX, 1, (0, 240, 255), 2)
    
    cv2.imshow("Muhammed Gazi - AI Ses Kontrolu", img)
    
    # Çıkış Kontrolü (Klavyeden 'q' harfine basılırsa)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
